## API

```yaml
kind: HTTPRoute
apiVersion: gateway.networking.k8s.io/v1alpha2
metadata:
  name: http-filter-1
spec:
  rules:
    - matches:
      - path:
          type: Prefix
          value: /
      filters:
      - type: RequestRewrite
        requestRewrite:
          path:
            type: Prefix
            pattern: /foo
            substitution: /bar
```


```go
// HTTPRequestRewrite defines configuration for the RequestRewrite filter.
type HTTPRequestRewrite struct {
    // Hostname is the value to be used to replace the Host header value
    // during forwarding.
    //
    // Support: Extended
    //
    // +optional
    Hostname *string `json:"hostname,omitempty"`

    // PathRewrite
    //
    // +optional
    PathRewrite *PathRewrite `json:"path,omitempty"`
}

type PathRewrite struct {
    // Type indicates the type of rewrite this is. Valid options include:
    //
    // * Prefix (Support: Extended)
    // * RegularExpression (Support: Custom)
    //
    // +optional
    // +kubebuilder:default=Prefix
    Type *string `json:"type,omitempty"`

    // Pattern specifies the value that should be rewritten. This may be omitted
    // for Prefix rewrites when
    //
    // Support: Extended
    //
    // +optional
    Pattern *string `json:"pattern"`

    // Path is the path matching
    //
    // Support: Extended
    Substitution *string `json:"path,omitempty"`
}
```
